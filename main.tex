\textbf{Assigned to:} Raman

  \item Decode \texttt{SYNT\{} with ROT13 to get \texttt{FLAG\{}.
  \item Apply Caesar shift 7 left on \texttt{3UJYFWA1VU\_4UK\_3UJ0K} to get \texttt{3NCRYPT1ON\_4ND\_3NC0D}.
  \item Base64-decode \texttt{M05DMERJTg==} to get \texttt{3NC0DIN}.
  \item Apply Atbash on \texttt{T\_ZIV\_} to get \texttt{G\_ARE\_}.
  \item XOR binary bytes to get \texttt{FUN\}}.
  \item Merge parts in order and resolve overlap (\texttt{3NC0D} + \texttt{3NC0DIN} $\rightarrow$ \texttt{3NC0DING}).
\end{enumerate}
