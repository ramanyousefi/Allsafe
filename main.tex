\subsection*{1) Multi-step Transformations (EASY, 5 points)}
\textbf{Assigned to:} Raman

\textbf{Steps}
\begin{enumerate}[leftmargin=*,noitemsep]
  \item Decode \texttt{SYNT\{} with ROT13 to get \texttt{FLAG\{}.
  \item Apply Caesar shift 7 left on \texttt{3UJYFWA1VU\_4UK\_3UJ0K} to get \texttt{3NCRYPT1ON\_4ND\_3NC0D}.
  \item Base64-decode \texttt{M05DMERJTg==} to get \texttt{3NC0DIN}.
  \item Apply Atbash on \texttt{T\_ZIV\_} to get \texttt{G\_ARE\_}.
  \item XOR binary bytes to get \texttt{FUN\}}.
  \item Merge parts in order and resolve overlap (\texttt{3NC0D} + \texttt{3NC0DIN} $\rightarrow$ \texttt{3NC0DING}).
\end{enumerate}

\textbf{Flag:} \texttt{FLAG\{3NCRYPT1ON\_4ND\_3NC0DING\_ARE\_FUN\}}

\textbf{Assigned to:} Raman

\subsection*{2) Last Broadcast (EASY)}
\textbf{Assigned to:} Filippo

\textbf{Steps}
\begin{enumerate}[leftmargin=*,noitemsep]
  \item Check the environment and notice that \texttt{PATH} points to trojanized binaries in \texttt{/opt/bunker/bin}.
  \item Bypass the fake tools by using absolute paths such as \texttt{/bin/ls} and \texttt{/bin/cat}.
  \item Read \texttt{/srv/terminal/.cache/.last\_broadcast} and capture the flag.
\end{enumerate}

\textbf{Flag:} \texttt{FLAG\{BUNKER-81071-bjgiap.esvmrvhq.pznmtce\}}

\textbf{Notes}
\begin{itemize}[leftmargin=*,noitemsep]
  \item This challenge demonstrates command hijacking through a malicious \texttt{PATH} configuration.
  \item Using absolute paths is a simple but effective way to bypass fake wrapper scripts in compromised environments.
\end{itemize}
